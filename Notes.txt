  method: 'POST',
  headers: {
    "Content-Type": "application/json",
    "XSRF-TOKEN": `<value of XSRF-TOKEN cookie>`
  },
  body: JSON.stringify({
    firstName: 'Peter',
    lastName: 'Parker',
    email: 'firestar@spider.man',
    username: 'Firestar',
    password: 'password'
  })
}).then(res => res.json()).then(data => console.log(data));
